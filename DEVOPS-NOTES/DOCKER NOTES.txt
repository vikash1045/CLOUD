Step 1:

Copy this in notepad
______________________________________
# Use official Ubuntu as the base image
FROM ubuntu:22.04

# Install Apache2
RUN apt-get update && \
   apt-get install apache2 -y

# Copy your static HTML page (optional)
# Uncomment the line below if you have index.html in the same folder
# COPY index.html /var/www/html/

# Expose port 80
EXPOSE 80

# Start Apache in foreground
CMD ["apachectl", "-D", "FOREGROUND"]     
________________________________________

Must change this
 rename command  : Rename-Item -Path "dockerfile.txt" -NewName "Dockerfile"

  command :      docker build -t [image name] .

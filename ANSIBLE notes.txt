- name: Install and Start Nginx
  hosts: webservers
  become: yes
  gather_facts: no


  tasks:
    - name: Install Nginx
      apt:
        name: nginx
        state: present
        update_cache: yes

    - name: Ensure Nginx is running
      service:
        name: nginx
        state: started
        enabled: yes
